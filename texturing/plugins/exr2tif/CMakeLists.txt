PROJECT(EXR2TIF)

INCLUDE(${PLUGIN_COMMON_SOURCE_DIR}/local.cmake)
SET(EXR2TIF_SRCS ${PROJECT_SOURCE_DIR}/exr2tif.cpp ${PLUGIN_COMMON_SRCS})

INCLUDE_DIRECTORIES(${AQSIS_OPENEXR_INCLUDE_DIR})

INSTALL(TARGETS exr2tif 
	RUNTIME DESTINATION ${PLUGINDIR}
	LIBRARY DESTINATION ${PLUGINDIR}
	ARCHIVE DESTINATION ${PLUGINDIR}
	)
