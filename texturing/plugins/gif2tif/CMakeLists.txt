PROJECT(GIF2TIF)

INCLUDE(${PLUGIN_COMMON_SOURCE_DIR}/local.cmake)
SET(GIF2TIF_SRCS ${PROJECT_SOURCE_DIR}/gif2tif.c ${PLUGIN_COMMON_SRCS})

INCLUDE_DIRECTORIES(${AQSIS_TIFF_INCLUDE_DIR})

ADD_LIBRARY(gif2tif SHARED ${GIF2TIF_SRCS})
TARGET_LINK_LIBRARIES(gif2tif ${AQSIS_TIFF_LIBRARIES} ${AQSIS_ZLIB_LIBRARIES})

INSTALL(TARGETS gif2tif 
	RUNTIME DESTINATION ${PLUGINDIR}
	LIBRARY DESTINATION ${PLUGINDIR}
	ARCHIVE DESTINATION ${PLUGINDIR}
	)
