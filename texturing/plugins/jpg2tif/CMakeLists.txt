PROJECT(JPG2TIF)

INCLUDE(${PLUGIN_COMMON_SOURCE_DIR}/local.cmake)
SET(JPG2TIF_SRCS ${PROJECT_SOURCE_DIR}/jpg2tif.c ${PLUGIN_COMMON_SRCS})

INCLUDE_DIRECTORIES(${AQSIS_TIFF_INCLUDE_DIR} ${AQSIS_JPEG_INCLUDE_DIR})

ADD_LIBRARY(jpg2tif SHARED ${JPG2TIF_SRCS})
TARGET_LINK_LIBRARIES(jpg2tif ${AQSIS_TIFF_LIBRARIES} ${AQSIS_ZLIB_LIBRARIES} ${AQSIS_JPEG_LIBRARIES})

INSTALL(TARGETS jpg2tif 
	RUNTIME DESTINATION ${PLUGINDIR}
	LIBRARY DESTINATION ${PLUGINDIR}
	ARCHIVE DESTINATION ${PLUGINDIR}
	)
