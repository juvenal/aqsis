PROJECT(AQSLTELL)

FILE(GLOB AQSLTELL_SRCS ${AQSLTELL_SOURCE_DIR}/*.cpp)
FILE(GLOB AQSLTELL_HDRS ${AQSLTELL_SOURCE_DIR}/*.h)

INCLUDE(${AQSISTYPES_SOURCE_DIR}/export.cmake)
INCLUDE(${ARGPARSE_SOURCE_DIR}/export.cmake)
INCLUDE(${SLXARGS_SOURCE_DIR}/export.cmake)
INCLUDE(${AQSISTEX_SOURCE_DIR}/export.cmake)

ADD_DEFINITIONS(-DDEFAULT_SHADER_PATH="\\"/usr/local/aqsis/share/shaders\\"")

ADD_EXECUTABLE(aqsltell ${AQSLTELL_SRCS} ${AQSLTELL_HDRS})

SET(LINKLIBS ${AQSLTELL_SYSTEM_LINKLIBS})
INCLUDE(${ARGPARSE_SOURCE_DIR}/link.cmake)
INCLUDE(${AQSISTYPES_SOURCE_DIR}/link.cmake)
INCLUDE(${SLXARGS_SOURCE_DIR}/link.cmake)
INCLUDE(${SHADERVM_SOURCE_DIR}/link.cmake)
INCLUDE(${AQSISTEX_SOURCE_DIR}/link.cmake)
TARGET_LINK_LIBRARIES(aqsltell ${LINKLIBS})

SET_TARGET_PROPERTIES(aqsltell PROPERTIES INSTALL_RPATH "${CMAKE_INSTALL_PREFIX}/${RENDERENGINEDIR}")
INSTALL(TARGETS aqsltell
	RUNTIME DESTINATION ${BINDIR})

SOURCE_GROUP("Header Files" FILES ${AQSLTELL_HDRS})
